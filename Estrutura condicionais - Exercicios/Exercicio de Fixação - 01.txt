// Elabore um algoritmo que leia dois números e mostre a soma deles.

programa {
  funcao inicio() {
    inteiro n1 
    inteiro n2
    inteiro soma

    escreva("Digite o n1:")
    leia(n1)
    escreva("\n")

    escreva("Digite o n2:")
    leia(n2)
    escreva("\n")

    soma = n1 + n2

    escreva("A soma dos valores é ")
    escreva(soma)
    escreva("\n")   
    
  }
}

// Construa um algoritmo que leia 4 notas e mostre a média.

programa {
  funcao inicio() {
    real nota1
    real nota2
    real nota3
    real nota4

    real media

    leia(nota1)
    leia(nota2)
    leia(nota3)
    leia(nota4)

    media = (nota1 + nota2 + nota3 + nota4) /4

    escreva("A média das notas é: ")
    escreva(media)

    
  }
}


// Desenvolva um algoritmo que receba dois valores numéricos inteiros, 
// calcule e mostre a soma do quadrado desses dois números.

programa {
  funcao inicio() {
    inteiro n1
    inteiro n2

    inteiro somaQuadradoDoisNumeros 

    leia(n1)
    leia(n2)

    somaQuadradoDoisNumeros = (n1)*(n1) + (n2)*(n2)

    escreva(somaQuadradoDoisNumeros)
    
  }
}
