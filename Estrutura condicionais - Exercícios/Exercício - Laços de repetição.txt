// Escreva um algoritmo que leia dez valores numéricos inteiros e apresente na tela o somatório dos valores.

programa {
  funcao inicio() {
    inteiro valor = 0
    inteiro soma = 0 
    inteiro cont = 0

    para (cont = 1; cont <=10; cont++){
      leia(valor)
      soma = soma + valor

      escreva("cont: ", cont, " valor: " ,valor, " soma: ", soma, "\n" )
    }

    escreva(soma)
    
  }
}
